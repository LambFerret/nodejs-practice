CREATE TABLE USER ( 
    UserID varchar(30) UNIQUE NOT NULL,
    UserNM varchar(50) NOT NULL,
    UserPw varchar(50) NOT NULL,
    UserEmail varchar(50) UNIQUE NOT NULL,
    PRIMARY KEY (UserID)
); 

CREATE TABLE UploadImg (
    Up_Img_ID int UNIQUE NOT NULL,
    UserID varchar(30) NOT NULL,
    Up_Img_Nm varchar(50) UNIQUE NOT NULL,
    PRIMARY KEY (Up_Img_ID),
    FOREIGN KEY (UserID) REFERENCES USER(UserID)
); 

CREATE TABLE Conv_Info (
    Conv_ID int UNIQUE NOT NULL,
    Original_Img_Type char(20) NOT NULL,
    Conv_Img_Type char(20) NOT NULL,
    PRIMARY KEY (Conv_ID)
);

CREATE TABLE Conv_Img (
    Conv_Img_ID int UNIQUE NOT NULL,
    Up_Img_Nm int UNIQUE NOT NULL,
    Conv_ID int NOT NULL,
    Conv_Img_Nm char(50) NOT NULL,
    PRIMARY KEY (Conv_Img_ID),
    FOREIGN KEY (Up_Img_Nm) REFERENCES UploadImg(Up_Img_ID),
    FOREIGN KEY (Conv_ID) REFERENCES Conv_Info(Conv_ID)
);

CREATE TABLE Posting (
    PostID int UNIQUE NOT NULL,
    Post_Text varchar(255),
    Post_Time timestamp NOT NULL,
    PostBoard_Type char NOT NULL,
    PRIMARY KEY (PostID),
    FOREIGN KEY (PostID) REFERENCES Conv_Img(Conv_Img_ID)
);
